C# 笔记：反射

發表於 2020-11-21 更新於 2020-11-22 分類於 C# 笔记

简介：

　　顾名思义，是记录 反射(Reflection) 用法的 C# 笔记~
　　反射在《C# 程序设计教程（第3版）》中位于「5.7 属性」一节。

引用书中定义如下：

　　反射(Reflection)是 .NET 中重要的机制，通过反射可以在运行时获得 .NET 中每一个类型（包括类、结构、委托、接口、枚举等）的成员，包括方法、属性、事件、和构造函数等，还可以获得每个成员的名称、限定符和参数等。有了反射，编程人员即可对每一个类型了如指掌。如果获得了构造函数的信息，可以直接创建对象，即使这个对象的类型在编译时还不知道。

说明：

　　要用反射，记得要调用System.Reflection！
　　反射有两种，透过名字，和透过程序集(.dll)。
　　·Type：项目
　　　　获取：（透过名字，只有一个）
　　　　　　　Type <项目变量名> = <项目变量名>.GetType(<想反射项目的名字，如"System.Int32">)
　　　　　　　（透过程序集，一或多个）
　　　　　　　Type[] <项目数组名> = <程序集变量名>.GetTypes();
　　·Assembly：程序集，如果要放射.dll，要先赋值 Assembly，再获取 Type。
　　　　获取：Assembly <程序集变量名> = Assembly.LoadFrom(<.dll档位址>);
　　·MemberInfo：项目成员
　　　　获取：（指定一个）
　　　　　　　MemberInfo <成员变量名> = <项目变量名>.GetMember(<成员名>);
　　　　　　　（全部）
　　　　　　　MemberInfo[] <成员数组名> = <项目变量名>.GetMembers();
　　·MethodInfo：项目方法
　　　　获取：（指定一个）
　　　　　　　MethodInfo <方法变量名> = <项目变量名>.GetMethod(<方法名>);
　　　　　　　（全部）
　　　　　　　MethodInfo[] <方法数组名> = <项目变量名>.GetMethods();
　　·FieldInfo：项目字段
　　　　获取：（指定一个）
　　　　　　　FieldInfo <字段变量名> = <项目变量名>.GetField(<字段名>);
　　　　　　　（全部）　　
　　　　　　　FieldInfo[] <字段数组名> = <项目变量名>.GetFields();
　　·ConstructorInfo：项目构造函数
　　　　获取：（指定一个）
　　　　　　　ConstructorInfo <构造函数变量名> = <项目变量名>.GetConstructor(<构造函数名>);
　　　　　　　（全部）　　
　　　　　　　ConstructorInfo[] <构造函数数组名> = <项目变量名>.GetConstructors();
　　·PropertyInfo：项目属性
　　　　获取：（指定一个）
　　　　　　　PropertyInfo <属性变量名> = <项目变量名>.GetProperty(<属性名>);
　　　　　　　（全部）　　
　　　　　　　PropertyInfo[] <属性数组名> = <项目变量名>.GetPropertys();
　　·EventInfo：项目事件
　　　　获取：（指定一个）
　　　　　　　EventInfo <事件变量名> = <项目变量名>.GetEvent(<事件名>);
　　　　　　　（全部）　　
　　　　　　　EventInfo[] <事件数组名> = <项目变量名>.GetEvents();
　　·ParameterInfo：方法参数
　　　　获取：（全部）　　
　　　　　　　ParameterInfo[] <参数数组名> = <方法变量名>.GetParameters();

实例：

using

using System;
using System.Reflection; //别忘了加

class HB_Reflection { } 普通反射

class HB_Reflection {
    protected string Classname; //项目名称
    protected Type Type; //项目
    protected MemberInfo[] Mmi { //项目的成员数组
        get { //这里就运用了属性
            if (Type == null) return null;
            else return Type.GetMembers(); //获取项目的成员
        }
    }
    protected MethodInfo[] Mti { //项目的方法数组
        get {
            if (Type == null) return null;
            else return Type.GetMethods();
        }
    }
    protected FieldInfo[] Fi { //项目的字段数组，即常量
        get {
            if (Type == null) return null;
            else return Type.GetFields();
        }
    }
    protected ConstructorInfo[] Ci { //项目的构造函数数组
        get {
            if (Type == null) return null;
            else return Type.GetConstructors();
        }
    }
    protected PropertyInfo[] Pi { //不是3.14蛤，是项目的属性数组
        get {
            if (Type == null) return null;
            else return Type.GetProperties();
        }
    }
    protected EventInfo[] Ei { //项目的事件数组
        get {
            if (Type == null) return null;
            else return Type.GetEvents();
        }
    }
    public Reflection(string name) { //构造函数
        Classname = name;
        Type = Type.GetType(name);
    }
    protected void Show() { //显示反射结果
        //如果Type == null，就是反射错误了。
        if (Type == null) Console.WriteLine("「{0}」：反射错误！", Classname); 
        else {
            Console.WriteLine("「{0}」类共有 {1} 个「成员」", Classname, Mmi.Length);
            if (Mti.Length != 0) {
                Console.WriteLine("  共 {0} 个「方法」，列表如下：", Mti.Length);
                foreach (MethodInfo item in Mti) {
                    Console.Write("    -{0}", item.Name);
                    ParameterInfo[] PaI = item.GetParameters(); //方法要多少参数
                    if (PaI.GetLength(0) != 0) {
                        Console.WriteLine("，需要 {0} 个参数", PaI.GetLength(0));
                        foreach (ParameterInfo pai in PaI) Console.Write("\t{0} {1}", pai.ParameterType, pai.Name);
                    }
                    Console.WriteLine();
                }
            }
            if (Fi.Length != 0) {
                Console.WriteLine("  共 {0} 个「字段」，列表如下：", Fi.Length);
                foreach (FieldInfo item in Fi) Console.WriteLine("    -{0}", item.Name);
            }
            if (Ci.Length != 0) {
                Console.WriteLine("  共 {0} 个「析构函数」，列表如下：", Ci.Length);
                foreach (ConstructorInfo item in Ci) Console.WriteLine("    -{0}", item.Name);
            }
            if (Pi.Length != 0) {
                Console.WriteLine("  共 {0} 个「属性」，列表如下：", Pi.Length);
                foreach (PropertyInfo item in Pi) Console.WriteLine("    -{0}", item.Name);
            }
            if (Ei.Length != 0) {
                Console.WriteLine("  共 {0} 个「事件」，列表如下：", Ei.Length);
                foreach (EventInfo item in Ei) Console.WriteLine("    -{0}", item.Name);
            }
            Console.WriteLine();
        }
    }
    public virtual void Display() {
        Show();
    }
    public virtual void Display(int page) {
        Show();
    }
}

Reflection_dll { } 程序集(.dll)反射

class Reflection_dll : Reflection { 
    public Assembly assembly; //加载程序集
    public Type[] types; //项目数组
    private int pages; //每10个项目为1页，分开显示。
    public Reflection_dll(string name) : base(name) {
        assembly = Assembly.LoadFrom(name);
        types = assembly.GetTypes();
        pages = types.GetLength(0) / 10 + 1;
    }
    public override void Display() { //显示
        if (pages == 1) {
            Console.WriteLine("「{0}」包含的 {1} 个项目名如下：", assembly.GetName().Name + ".dll", types.GetLength(0));
            for (int i = 0; i < types.GetLength(0); i++) {
                Console.Write("({0})", i + 1);
                Classname = types[i].FullName;
                Type = types[i];
                Show();
            }
        }
        else {
            Console.WriteLine("「{0}」共包含 {1} 个项目，每 10 个为 1 页，分成 {2} 页\n「{0}」的第 1 页项目名如下：", assembly.GetName().Name, types.GetLength(0), pages);
            for (int i = 0; i < 10; i++) {
                Console.Write("({0})", i + 1);
                Classname = types[i].FullName;
                Type = types[i];
                Show();
            }
        }
    }
    public override void Display(int page) { //显示指定页数
        int start = (page - 1) * 10;
        Console.WriteLine("「{0}」的第 {1} 页项目名如下：", assembly.GetName().Name, page);
        for (int i = start; i < start + 10 && i < types.GetLength(0); i++) {
            Console.Write("({0})", i + 1);
            Classname = types[i].FullName;
            Type = types[i];
            Show();
        }
    }
}

main调用（反射程序集）

程序集是从vs的输出窗口来的，表示任何正常的 .cs 档生成的 .dll 档都可以运行。

Reflection_dll redll = new Reflection_dll(@"C:\WINDOWS\Microsoft.Net\assembly\GAC_32\mscorlib\v4.0_4.0.0.0__b77a5c561934e089\mscorlib.dll");
redll.Display();
